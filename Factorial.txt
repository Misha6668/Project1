num = 14 

def factor(n):
    cet=0
    b=0
    if n == 0:
        return 1
        
    else:
        for i in range(1, n + 1, 1):
            if b != 0:
                cet = cet * i
            else:
                cet += i
                
            b += 1
            
    return cet
    
    
result = factor(num)

print(result)
