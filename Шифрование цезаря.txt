word = str(input('Enter word: '))
choose = int(input('Encode or decode\n1 - encode\n2 - decode: '))

def fun(w, cho):
    w = list(w)
    new_w = []
    for i in w:
        if ord(i) < 97:
            new_w.append(chr(ord(i) + 32))
        else:
            new_w.append(i)
    en = 'abcdefghijklmnopqrstuvwxyz'
    en2 = 'cdefghijklmnopqrstuvwxyzab'
    en = list(en)
    en2 = list(en2)
    res = ''
    if cho == 1:
        for i in new_w:
            if i in en:
                res += en2[en.index(i)]
            else:
                return 'Error'
    else:
        for el in new_w:
            if el in en2:
                res += en[en2.index(el)]
            else:
                return 'Error'
                
    return res
    
result = fun(word,choose)

print(result)
