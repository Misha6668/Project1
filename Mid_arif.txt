numbers = ((5, 4, 5, 4), (3, 3, 4, 6), (8, 9, 5, 4), (12, 4, 5, 1), (9, 3, 5, 1))

def mid(tpl1):
    for i in tpl1:
        mid = sum(tpl1) / len(tpl1)
        
    return mid
    
def fun(tpl):
    mass = []
    for i in tpl:
        mass.append(mid(i))
        
    return mass
    
result = fun(numbers)

print(result)
